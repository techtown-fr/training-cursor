% TechTown PDF Template for Pandoc
% Uses only basictex-compatible packages

\usepackage{fancyhdr}
\usepackage{graphicx}
\usepackage{xcolor}

% TechTown brand colors
\definecolor{ttblue}{HTML}{1c63ed}
\definecolor{ttdarkblue}{HTML}{1557d6}
\definecolor{ttaccent}{HTML}{3b7eff}
\definecolor{tttext}{HTML}{1f2937}
\definecolor{ttgray}{HTML}{6b7280}
\definecolor{ttlightbg}{HTML}{f0f4ff}
\definecolor{ttcodebg}{HTML}{f3f4f6}

% Set text color
\color{tttext}

% Links in TechTown blue (deferred until hyperref is loaded by pandoc)
\AtBeginDocument{%
  \hypersetup{
    colorlinks=true,
    linkcolor=ttblue,
    urlcolor=ttblue,
    citecolor=ttblue
  }%
}

% Header and footer
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0.5pt}
\renewcommand{\headrule}{\hbox to\headwidth{\color{ttblue}\leaders\hrule height \headrulewidth\hfill}}
\renewcommand{\footrulewidth}{0.3pt}
\renewcommand{\footrule}{\hbox to\headwidth{\color{ttgray}\leaders\hrule height \footrulewidth\hfill}}

\fancyhead[L]{\includegraphics[height=18pt]{techtown-logo.png}}
\fancyfoot[L]{\small\color{ttgray}Formation TechTown}
\fancyfoot[R]{\small\color{ttgray}\thepage}

% Add space between header and page content
\setlength{\headsep}{40pt}

% Style section headings with TechTown blue
\makeatletter
\renewcommand{\section}{\@startsection{section}{1}{0pt}%
  {-2ex plus -1ex minus -0.2ex}%
  {1.5ex plus 0.2ex}%
  {\normalfont\Large\bfseries\color{ttblue}}}

\renewcommand{\subsection}{\@startsection{subsection}{2}{0pt}%
  {6ex plus 1.5ex minus 0.5ex}%
  {1.5ex plus 0.2ex}%
  {\normalfont\large\bfseries\color{ttdarkblue}}}

\renewcommand{\subsubsection}{\@startsection{subsubsection}{3}{0pt}%
  {3ex plus 0.5ex minus 0.2ex}%
  {0.8ex plus 0.1ex}%
  {\normalfont\normalsize\bfseries\color{ttaccent}}}
\makeatother

% Custom title page
\makeatletter
\renewcommand{\maketitle}{%
  \thispagestyle{empty}
  \vspace*{2cm}
  \begin{center}
    \includegraphics[height=50pt]{techtown-logo.png}

    \vspace{2cm}

    {\Huge\bfseries\color{ttblue}\@title\par}

    \vspace{0.8cm}

    \ifx\@subtitle\undefined\else
      {\Large\color{ttgray}\@subtitle\par}
    \fi

    \vspace{1.5cm}

    {\large\color{tttext}\@author\par}

    \vspace{0.5cm}

    {\normalsize\color{ttgray}\@date\par}

    \vspace{2cm}

    {\color{ttblue}\rule{0.6\textwidth}{1pt}}
  \end{center}
  \newpage
}

% Support subtitle from pandoc
\providecommand{\subtitle}[1]{%
  \def\@subtitle{#1}
}
\makeatother

% Code block styling
\let\oldverbatim\verbatim
\let\endoldverbatim\endverbatim
\renewenvironment{verbatim}{%
  \par\noindent
  \begin{oldverbatim}%
}{%
  \end{oldverbatim}%
  \par
}
